\section*{One-paragraph summary}
The paper develops a pricing framework for derivatives whose payoff
depends on a Constant Maturity Treasury (CMT) yield, taking default
risk of the bond issuer into account. The key conceptual move is to
identify $\Rcmt_t$ with a \emph{risky forward swap rate with recovery}
on a credit-risk-adjusted curve, so that it is a ratio of tradable
assets and therefore a martingale under the corresponding annuity
(swaption) measure. The naive risky swap rate has a singularity at
default; the partial filtration / credit-risk-adjusted framework
(Bielecki--Jeanblanc--Rutkowsky, Jamshidian) sidesteps this. Three
versions of a CMT-let payoff are priced (A: survival to payment date;
B: survival to fixing; C: B plus pre-default value if default precedes
fixing), and closed-form convexity corrections are obtained under a
one-factor linear swap-rate model with lognormal $\Rcmt$ dynamics. In
the no-default limit, the results collapse to Pelsser's classical
convexity correction.

\section{Setup and notation}

The paper distinguishes carefully between three discount-bond objects:
risk-free $\Disc{t}{T}$, risky $\widetilde{\Disc{t}{T}}$ (zero at
default), and \emph{credit-risk-adjusted / pre-default}
$\widehat{\Disc{t}{T}}$ (the analytically convenient version that
remains continuous through default). The risky annuity $\widehat{\Ann{t}}$
is the natural numeraire for $\Rcmt$.

\section{Key equations}

\paragraph{Bond price as function of YTM} (compounded convention):
\begin{equation}\label{eq:cmt-ytm}
  \Bond{T_s}{} \;=\; \prod_{i=1}^{n} \frac{1}{1 + \alpha_i R^{ytm}_{T_s}}
                  \;+\; K \sum_{i=1}^n \alpha_i \prod_{j=1}^i \frac{1}{1 + \alpha_j R^{ytm}_{T_s}}.
\end{equation}
\textit{Says:} same compounded form as in Pucci's T-Lock paper; the
YTM is a non-linear function of the bond price.

\paragraph{Definition of $\Rcmt$ (par-coupon on the risky curve).}
\begin{equation}\label{eq:cmt-def}
  1 \;=\; R^{cmt}_{T_s} \widehat A_{T_s} \;+\; \widehat D_{T_s T_n} \;+\; \Rec\, [\text{default leg}]_{T_s}.
\end{equation}
\textit{Says:} $\Rcmt_{T_s}$ is the fixed rate that prices a risky
coupon bond at par, with the three summands being coupon PV, notional
PV, and the recovery default leg. Equivalent definitions on the
credit-risk-adjusted curve sidestep the default-time discontinuity.

\paragraph{Link between YTM and CMT} (Proposition 2.1):
\begin{equation}\label{eq:ytm-cmt}
  R^{ytm}(B, K) \;\cong\; R^{cmt} + (K - R^{cmt}) - \frac{R^{cmt}}{1 - (1+R^{cmt})^{-n}} (B - 1)
                \;\cong\; R^{cmt} + (K - R^{cmt}) - \frac{B - 1}{n}.
\end{equation}
\textit{Says:} first-order Taylor expansion of $R^{ytm}$ around the
``at par'' point $(B, K) = (1, R^{cmt})$. The expansion linearises YTM
in $B$ and $K$, so CMY pricing on the risky curve reduces (to first
order) to a CMT-on-risky-swap-rate problem.

\paragraph{Three CMT-let payoffs} (Defs.\ 3.1--3.3, payment at $T_p \ge T_s$):
\begin{align}
  \text{(A)} \quad & \Ind_{[\tau > T_p]}\, R^{ytm}_{T_s}, \label{eq:payoff-A}\\
  \text{(B)} \quad & \Ind_{[\tau > T_s]}\, R^{ytm}_{T_s}, \label{eq:payoff-B}\\
  \text{(C)} \quad & \Ind_{[\tau > T_s]}\, R^{ytm}_{T_s} \;+\; \Ind_{[\tau \le T_s]}\, R^{ytm}_{\tau-}. \label{eq:payoff-C}
\end{align}
\textit{Says:} (A) is a survival-contingent payoff (no payment if
default occurs in $[T_s, T_p]$); (B) is contingent on survival to the
fixing date only; (C) is non-contingent: if default occurs before
fixing, pays the last available fixing. NPV ordering:
$\NPV^{(C)} \ge \NPV^{(B)} \ge \NPV^{(A)}$.

\paragraph{Convexity correction (definition):}
\begin{equation}\label{eq:cc-def}
  CC^{(i)} \;\equiv\; \frac{\NPV^{(i)}}{D \cdot F} - 1, \qquad i \in \{A, B, C\},
\end{equation}
where $F$ is the forward of the rate and $D$ the natural discount.
\textit{Says:} the convexity correction is what you add to the
``forward rate times discount'' deterministic value to obtain the
true risk-neutral price.

\paragraph{One-factor linear swap-rate model} (Sec.\ 5):
\begin{equation}\label{eq:linear-G}
  \frac{\widehat D_{T_s U}}{\widehat A_{T_s}} \;=\; G_U(\Rcmt_{T_s}) \;=\; \alpha + \beta_U \Rcmt_{T_s},
\end{equation}
for all $U \in \{T_p, T_1, \ldots, T_n\}$, with
\begin{equation}\label{eq:linear-calib}
  \alpha = \frac{1}{\sum_{i=1}^n \alpha_i}, \qquad
  \beta_U = \frac{\chi_0 \widehat D_{0U} / \widehat A_0 - \alpha}{\Rcmt_0}.
\end{equation}
\textit{Says:} the ratio of risky DF to risky annuity is modelled as a
linear function of $\Rcmt$. Calibration is forced by $\Rcmt_0 \to 0$
limit (for $\alpha$) and martingale property under the annuity measure
(for $\beta_U$).

\paragraph{Lognormal CMT dynamics:}
\begin{equation}\label{eq:lognormal-dyn}
  \frac{\dd \Rcmt_t}{\Rcmt_t} \;=\; \sigma \, \dd W^{\widehat A}_t.
\end{equation}
\textit{Says:} $\Rcmt$ is lognormal under the annuity measure, with
volatility $\sigma$. Combined with $W^{\widehat A}$ being a Brownian
motion under the enlarged filtration $\mathcal G$, this gives
closed-form expectations.

\paragraph{Closed-form convexity corrections} (Proposition 6.1, under DI + lognormal; AC + constant intensity $\gamma$ for (C)):
\begin{align}
  CC^{(A)}_0 \;=\; CC^{(B)}_0 &= \bigl(e^{\sigma^2 T_s} - 1\bigr) \left(1 - \frac{\alpha \widehat A_0}{\chi_0 \widehat D_{0 T_p}}\right), \label{eq:cc-AB}\\
  CC^{(C)}_0 &= \bigl(1 + CC^{(B)}_0\bigr) \frac{\widehat D_{0 T_s}}{D_{0 T_s}} - 1
              + \frac{\alpha \widehat A_0}{\chi_0 \widehat D_{0 T_p}}\bigl(1 - e^{-\gamma T_s}\bigr)
              + \left(1 - \frac{\alpha \widehat A_0}{\chi_0 \widehat D_{0 T_p}}\right) \frac{e^{(\sigma^2 - \gamma) T_s} - 1}{\sigma^2 - \gamma}\, \gamma. \label{eq:cc-C}
\end{align}
\textit{Says:} clean closed forms. $CC^{(A)} = CC^{(B)}$ holds because
the survival difference between the two payoffs is exactly absorbed
by the credit-risk-adjusted DF ratio in the discount. $CC^{(C)}$ has
three distinct contributions: rate convexity (B), default-driven jump
to pre-default value, and a $\gamma$-weighted integral. All vanish at
$\sigma = 0$ (verified explicitly in the paper).

\paragraph{No-default limit} (Sec.\ 6, $\gamma \to 0$):
\begin{equation}\label{eq:no-default-limit}
  CC^{(A)}_0 \;=\; CC^{(B)}_0 \;=\; CC^{(C)}_0
  \;\longrightarrow\; \bigl(e^{\sigma^2 T_s} - 1\bigr)\left(1 - \frac{\alpha A_0}{D_{0 T_p}}\right),
\end{equation}
\textit{Says:} recovers the classical Pelsser (2001) convexity
correction for a CMS-let under the linear swap-rate model.

\section{Derivations \& commentary}

\begin{commentary}
The key structural move is identifying $\Rcmt$ with the risky forward
swap rate with recovery: in the appendix Pucci shows
$\Rcmt_{T_s} = \widehat R^{swp}_{T_s}$ at fixing. This buys two things:
(i) $\Rcmt$ is a ratio of tradable assets, hence a martingale under
the credit-risk-adjusted annuity measure $\widehat A$; (ii) the
machinery from CMS pricing (Hagan, Hunt--Kennedy linear swap-rate
model) transfers almost verbatim, with $\widehat D$ and $\widehat A$
in place of $D$ and $A$.
\end{commentary}

\begin{commentary}
The three payoff variants (A/B/C) matter operationally because they
correspond to different contract designs. (A) and (B) are commonly
embedded in survival-contingent structures (e.g.\ a swap that
cancels on credit event). (C) is the building block for
non-contingent CMT-linked notes. The ordering $\NPV^{(C)} \ge
\NPV^{(B)} \ge \NPV^{(A)}$ is structural, not assumption-dependent.
\end{commentary}

\begin{commentary}
The Taylor link \eqref{eq:ytm-cmt} is a small but useful result on its
own: it says that to first order in $(B-1)$ and $(K - R^{cmt})$, the
non-linear YTM equals an affine function of CMT. So once you can price
CMT-contingent payoffs, you can approximate CMY-contingent payoffs by
the same formula plus a known linear correction. The expansion
deteriorates for bonds far from par; for ARM-mortgage indexing
applications (the paper's motivating use case) the bonds are near par
and the approximation is tight.
\end{commentary}

\begin{commentary}
The $CC^{(C)}$ formula has a special structure worth noting: the
$\gamma \to \sigma^2$ limit is regular (the $(\sigma^2 - \gamma)^{-1}$
factor cancels against the exponential), and the
$\gamma \to 0$ limit collapses all three corrections to
Pelsser. So the closed forms degrade smoothly as default risk
disappears --- a desirable property for a risky CMT framework.
\end{commentary}

\section{Validation}

\begin{validation}
This is a \emph{specification} of mathematical objects the paper
defines and the numerical values they should take. It says nothing
about how those objects are implemented or invoked.

\textbf{Quantities the paper defines:}
\begin{itemize}[noitemsep]
  \item Bond price as function of YTM (eq.~\ref{eq:cmt-ytm}), and its
        inverse $R^{ytm}(B, K)$ defined implicitly.
  \item $\Rcmt_{T_s}$ as the risky par-coupon rate defined by
        \eqref{eq:cmt-def}.
  \item The YTM--CMT link \eqref{eq:ytm-cmt} (first-order Taylor).
  \item Three CMT-let payoffs $\NPV^{(A,B,C)}$ given by
        \eqref{eq:payoff-A}--\eqref{eq:payoff-C}.
  \item Linear swap-rate model coefficients $\alpha, \beta_U$ from
        \eqref{eq:linear-calib}.
  \item Closed-form convexity corrections $CC^{(A)}, CC^{(B)}, CC^{(C)}$
        in \eqref{eq:cc-AB}--\eqref{eq:cc-C}.
\end{itemize}

\textbf{Canonical numerical case:} the paper provides no worked
numerical example in the body. Suggested test bench: a 10Y CMT-let
on a par bond with $T_s = 5$Y, $T_p = T_s + 6$M, annual coupon (so
$n = 10$, $\alpha_i = 1$), flat hazard $\gamma$ across
$\{0\%, 1\%, 3\%\}$, lognormal vol $\sigma \in \{10\%, 20\%\}$,
flat risk-free curve at $2\%$.

\textbf{Expected numerical values:} no quantitative benchmarks
printed in the paper. Validation is by closed-form vs.\ Monte Carlo
simulation of the lognormal dynamics, plus limiting-case identities
below.

\textbf{Invariants and identities to verify:}
\begin{enumerate}[noitemsep]
  \item \emph{YTM-CMT consistency}: at $B = 1$ and $K = R^{cmt}$,
        $R^{ytm}(B, K) = R^{cmt}$ exactly. (Pin point of the Taylor
        expansion.)
  \item \emph{$CC^{(A)} = CC^{(B)}$ identity}: the equality in
        \eqref{eq:cc-AB} should hold for any valid set of inputs
        under DI, not just at the canonical case.
  \item \emph{Vanishing at zero vol}: $CC^{(A)} = CC^{(B)} = 0$ at
        $\sigma = 0$. $CC^{(C)}$ should also vanish identically at
        $\sigma = 0$ under DI + AC + constant $\gamma$; this is
        verified algebraically in the paper.
  \item \emph{No-default limit}: as $\gamma \to 0$ (i.e.\
        $\widehat A \to A$, $\widehat D \to D$, $\chi \to 1$), all
        three corrections collapse to the Pelsser
        single-curve form $(e^{\sigma^2 T_s} - 1)(1 - \alpha A_0 / D_{0T_p})$.
  \item \emph{NPV ordering}: $\NPV^{(C)} \ge \NPV^{(B)} \ge \NPV^{(A)}$
        for any feasible inputs.
  \item \emph{Regular limit at $\gamma = \sigma^2$}: $CC^{(C)}$ does
        not blow up at $\gamma = \sigma^2$; the integral in
        \eqref{eq:cc-C} has a removable singularity and the limiting
        value can be evaluated via L'H\^opital.
  \item \emph{Closed form vs.\ Monte Carlo}: simulate $\Rcmt$ under
        \eqref{eq:lognormal-dyn} together with a doubly-stochastic
        default time of intensity $\gamma$; the MC estimator of
        $CC^{(C)}$ should match \eqref{eq:cc-C} within statistical
        error.
\end{enumerate}
\end{validation}
